package com.ts.datasource;

import cn.ft.ckn.fastmapper.aspect.SqlActuatorAspect;
import cn.ft.ckn.fastmapper.config.FastMapperConfig;
import com.alibaba.druid.pool.DruidDataSource;
import org.springframework.boot.context.properties.ConfigurationProperties;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import javax.sql.DataSource;

/**
 * @author ckn
 * @date 2023/11/28
 */
@Configuration
public class DruidConfig {
    static {
        FastMapperConfig.isOpenSQLPrint = true;
        FastMapperConfig.addMapperExpander(SqlActuatorAspect.class);
    }

    @Bean
    @ConfigurationProperties(prefix = "spring.datasource")
    public DataSource getMyDruidDataSource() {
        return new DruidDataSource();
    }
}


<dependency>
       <groupId>cn.ft.ckn</groupId>
        <artifactId>fast-mapper</artifactId>
        <version>${FASTMAPPER.VERSION}</version>
</dependency>

        GenerateTemplateConfig config = new GenerateTemplateConfig();
